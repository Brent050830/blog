   a=int(input("请输入一个数字："))# 循环条件要改变！
   if a==num:# 猜对时候的判断
       flag=0
   else:# 猜错
       if a>num:
           print("猜的数字大了")
       else:
           print("猜的数字小了")
   n+=1
print(f"猜对了，你猜的次数为：{n}")
